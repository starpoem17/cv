\entrybig
	{\textbf{Nudge-Based Kiosk Intervention for Reducing Sugar-Sweetened \\ Beverage Intake}}{Sep 2022 -- Dec 2022}
	{Data Science Project Course, Hanyang University}{}
{\renewcommand{\labelitemii}{\textbullet}
\innerlist{
	\entry{Designed a default-option nudge intervention for fast-food kiosks to reduce sugar-sweetened beverage selection, grounding the project in Seoul public health data and behavioral-science literature.}
	\entry{Analyzed relationships among adolescent obesity, fast-food consumption, and soda intake using Pearson correlation, identifying strong positive associations.}
    \entry{Evaluated a kiosk-style survey simulation; chi-squared analysis showed beverage choice depended on the default setting, with 54.5\% of participants switching from SSB to diet soda under the simulated intervention.}
}
