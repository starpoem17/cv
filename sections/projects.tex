	{\textbf{Nudge-Based Kiosk Intervention for Reducing Sugar-Sweetened \\ Beverage Intake}}{Sep 2022 -- Dec 2022}
	{Data Science Project Course, Hanyang University}{}
{\renewcommand{\labelitemii}{\textbullet}
\innerlist{
	\entry{Designed a default-option nudge intervention for fast-food kiosks to reduce sugar-sweetened beverage selection, grounding the project in Seoul public health data and behavioral-science literature.}
